CREATE OR REPLACE PROCEDURE SAMPLESDEMOS(P_ID IN NUMBER)
AS 
V_VCH_CUSTOMER_NAME VARCHAR2(30);
BEGIN
SELECT CUSTOMER_NAME INTO V_VCH_CUSTOMER_NAME FROM CUSTOMERS_DEMO WHERE CUSTOMER_ID=P_ID;
DBMS_OUTPUT.PUT_LINE('ID: '||P_ID||', NAME: '||V_VCH_CUSTOMER_NAME);
END;

EXEC SAMPLESDEMOS(2);

CREATE OR REPLACE PACKAGE EXAMPLEDEMO AS PROCEDURE EXAMPLE(P_ID IN NUMBER);
END;


CREATE OR REPLACE PACKAGE BODY EXAMPLEDEMO AS PROCEDURE EXAMPLE(P_ID IN NUMBER)
AS 
V_VCH_CUSTOMER_NAME VARCHAR2(30);
BEGIN
SELECT CUSTOMER_NAME INTO V_VCH_CUSTOMER_NAME FROM CUSTOMERS_DEMO WHERE CUSTOMER_ID=P_ID;
DBMS_OUTPUT.PUT_LINE('ID: '||P_ID||', NAME: '||V_VCH_CUSTOMER_NAME);
END EXAMPLE;
END EXAMPLEDEMO;

EXEC EXAMPLEDEMO.EXAMPLE(2);


//////////////

CREATE OR REPLACE PROCEDURE EXAMPLES(P_ID IN NUMBER)
AS
V_VCH_CUSTOMER_NAME VARCHAR2(30);
CURSOR C_CURSOR IS SELECT CUSTOMER_NAME FROM CUSTOMERS_DEMO WHERE CUSTOMER_ID=P_ID;
BEGIN
OPEN C_CURSOR;
FETCH C_CURSOR INTO V_VCH_CUSTOMER_NAME;
DBMS_OUTPUT.PUT_LINE('ID: '||P_ID||', '||'NAME:'||V_VCH_CUSTOMER_NAME);
CLOSE C_CURSOR;
END;


EXEC EXAMPLES(1);
